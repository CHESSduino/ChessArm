char i2s( char* buf, int arc )
{
  int e=0;
  buf[0] = 0;
  if ( arc < 0 )
  {
    strcpy( buf, "-" );
    buf++;
    arc = 0-arc;
  }
  while ( 10<=arc )
  {
    char n = ' ';
    n = arc - (arc/10)*10;
    arc = arc/10;
    buf[e] = n + '0';
    e ++;
  }
    char n = ' ';
    n = arc - (arc/10)*10;
    arc = arc/10;
    buf[e] = n + '0';
    e ++;
   for ( int i = 0; i< e/2; i++ )
   {
      n = buf[i];
      buf[i] = buf[e-i-1];
      buf[e-i-1] = n;
   }
    buf[e] = 0;
}

void Metal2_movecharint( char c, int i, bool ret=0 )
{
  char buf[80];
  buf[0] = c;
  // strcpy( buf, "N  " );
  buf[1] = 0;
  i2s( buf+1, i );
  if ( ret )
    Metal2_SerialPrintBoth( buf, 1 );
  else
  {
    strcat( buf, " " );
    Metal2_SerialPrintBoth( buf );
  }  
}
