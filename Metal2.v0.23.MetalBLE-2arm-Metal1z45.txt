char i2s( char* buf, int arc )
{
  int e=0;
  buf[0] = 0;
  if ( arc < 0 )
  {
    strcpy( buf, "-" );
    buf++;
    arc = 0-arc;
  }
  while ( 10<=arc )
  {
    char n = ' ';
    n = arc - (arc/10)*10;
    arc = arc/10;
    buf[e] = n + '0';
    e ++;
  }
    char n = ' ';
    n = arc - (arc/10)*10;
    arc = arc/10;
    buf[e] = n + '0';
    e ++;
   for ( int i = 0; i< e/2; i++ )
   {
      n = buf[i];
      buf[i] = buf[e-i-1];
      buf[e-i-1] = n;
   }
    buf[e] = 0;
}

void Metal2_movecharint( char c, int i, bool ret=0 )
{
  char buf[80];
  buf[0] = c;
  // strcpy( buf, "N  " );
  buf[1] = 0;
  i2s( buf+1, i );
  if ( ret )
    Metal2_SerialPrintBoth( buf, 1 );
  else
  {
    strcat( buf, " " );
    Metal2_SerialPrintBoth( buf );
  }  
}

void Metal2_moveservo( int n, int arc )
{
  Metal2_movecharint( '#', 25 );
  Metal2_movecharint( 'G', 202);
  Metal2_movecharint( 'N', n);
  Metal2_movecharint( 'V', arc, 1 );
}

void init_Metal2_eage()
{
  if ( Metal2_eage[0].x != 0 ) return;

  {
    Metal2_eage[2*i+0].x = 7 * ( ( Metal2_squenceloop[7*i+7].x - Metal2_squenceloop[7*i+7+6].x ) / 6 ) + Metal2_squenceloop[7*i+7+6].x;

  }

char Metal2_pumped = ' ';
void Metal2_54mov( char x, char y, bool autopump, bool off ) {
  if ( x < -4 || x > 4 ||
       y < 0 || y > 5 ) return;
  // do the full board serv position.
  init_Metal2_eage();
  // got the servo positions.
  int servo_x,
      servo_y,
      servo_z,
      servo_f,
      pumpx = x - (-4),
      pumpy = 5 - y;
  if ( x == -4 || x == 4 )
  {
    if ( y == 5 ) return; // out of the servo range.
    x = ( x - (-4) )/8;
    y = 4 - y;
    servo_x = Metal2_eage[y*2+x].x;
    servo_y = Metal2_eage[y*2+x].y;
    servo_z = Metal2_eage[y*2+x].z + 22;
    servo_f = Metal2_squenceloop[0].f;
  } else {
    x = x - (-3);
    y = 5 - y;
    servo_x = Metal2_squenceloop[y*7+x].x;
    servo_y = Metal2_squenceloop[y*7+x].y;
    servo_z = Metal2_squenceloop[y*7+x].z + 22;
    servo_f = Metal2_squenceloop[y*7+x].f;    
  }
  // add off.
  if ( off )
  {
    servo_x += Metal2_54off[pumpy*9+pumpx].x;
    servo_y += Metal2_54off[pumpy*9+pumpx].y;
    servo_z += Metal2_54off[pumpy*9+pumpx].z;
  }
  // mov 2.
//  Serial.print( "mov" );
  Metal2_moveto( servo_x, servo_y, servo_z, servo_f );
  // check pump flag.
  if ( autopump )
  {
//    Serial.print( "pup" );
     if ( pieces2[pumpy*9+pumpx] != ' ' &&
          Metal2_pumped == ' ' )
     {  // pump it.
        delay( 1000 );
        Metal2_moveto( servo_x, servo_y, servo_z - 2 - 20, servo_f );
        delay( 1000 );
        Metal2_pump( 1 );
        delay( 1000 );
        Metal2_moveto( servo_x, servo_y, servo_z, servo_f );
        // record.
        Metal2_pumped = pieces2[pumpy*9+pumpx];
        pieces2[pumpy*9+pumpx] = ' ';
        // 2 arm exchange.
        if ( pumpy >= 0 && pumpy <= 1 &&
             pumpx >= 1 && pumpx <= 7 )
        {
          setpieces( 8-pumpx + (1-pumpy)*9, ' ' );
        }
     } else
     if ( Metal2_pumped != ' ' &&
          pieces2[pumpy*9+pumpx] == ' ' )
     {  // release it.
        delay( 1000 + 1000 );//1000 * sqrt( (5-pumpy)*(5-pumpy) + (4-pumpx)*(4-pumpx) ) );
        Metal2_pump( 0 );
        pieces2[pumpy*9+pumpx] = Metal2_pumped;
        // 2 arm exchange.
        if ( pumpy >= 0 && pumpy <= 1 &&
             pumpx >= 1 && pumpx <= 7 )
        {
          setpieces( 8-pumpx + (1-pumpy)*9, Metal2_pumped );
        }
        Metal2_pumped = ' ';
     }
  }
}

void movecharint( char c, int i, bool ret=0 )
{
  char buf[80];
  buf[0] = c;
  // strcpy( buf, "N  " );
  buf[1] = 0;
  i2s( buf+1, i );
  if ( ret )
    SerialPrintBoth( buf, 1 );
  else
  {
    strcat( buf, " " );
    SerialPrintBoth( buf );
  }  
}

}

void Metal_54mov( char x, char y, bool autopump, bool off ) {
  if ( x < -4 || x > 4 ||
       y < 0 || y > 5 ) return;
  // do the full board serv position.
  init_eage();
  // got the servo positions.
  int servo_x,
      servo_y,
      servo_z,
      servo_f,
      pumpx = x - (-4),
      pumpy = 5 - y;
  if ( x == -4 || x == 4 )
  {
    if ( y == 5 ) return; // out of the servo range.
    x = ( x - (-4) )/8;
    y = 4 - y;
    servo_x = eage[y*2+x].x;
    servo_y = eage[y*2+x].y;
    servo_z = eage[y*2+x].z;
    servo_f = squenceloop[0].f;
  } else {
    x = x - (-3);
    y = 5 - y;
    servo_x = squenceloop[y*7+x].x;
    servo_y = squenceloop[y*7+x].y;
    servo_z = squenceloop[y*7+x].z;
    servo_f = squenceloop[y*7+x].f;    
  }
  // add off.
  if ( off )
  {
    servo_x += Metal_54off[pumpy*9+pumpx].x;
    servo_y += Metal_54off[pumpy*9+pumpx].y;
    servo_z += Metal_54off[pumpy*9+pumpx].z;
  }
  // mov 2.
  moveto( servo_x, servo_y, servo_z, servo_f );
  if ( autopump )
  {
//    Serial.print( "pp1" );
     if ( pieces[pumpy*9+pumpx] != ' ' &&
          pumped == ' ' )
     {  // pump it.
        delay( 1000 );
        moveto( servo_x, servo_y, servo_z - 2 - 20, servo_f );
        delay( 1000 );
        pump( 1 );
        delay( 1000 );
        moveto( servo_x, servo_y, servo_z, servo_f );
        // record.
        pumped = pieces[pumpy*9+pumpx];
        if ( pumpy != 5 )
          pieces[pumpy*9+pumpx] = ' ';
        // 2 arm exchange.
        if ( pumpy >= 0 && pumpy <= 1 &&
             pumpx >= 1 && pumpx <= 7 )
        {
          pieces2[ 8-pumpx + (1-pumpy)*9 ] = pieces[pumpy*9+pumpx];
        }
     } else
     if ( pumped != ' ' &&
          pieces[pumpy*9+pumpx] == ' ' )
     {  // release it.
        delay( 1000 + 1000 );//1000 * sqrt( (5-pumpy)*(5-pumpy) + (4-pumpx)*(4-pumpx) ) );
        pump( 0 );
        pieces[pumpy*9+pumpx] = pumped;
        // 2 arm exchange.
        if ( pumpy >= 0 && pumpy <= 1 &&
             pumpx >= 1 && pumpx <= 7 )
        {
          pieces2[ 8-pumpx + (1-pumpy)*9 ] = pumped;
        }
        pumped = ' ';
     }
  }
}
